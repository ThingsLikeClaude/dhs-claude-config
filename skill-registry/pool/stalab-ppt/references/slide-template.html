Code Example

// Presentation Engine
function goToSlide(idx, dir) {
  const cur = slides[currentIndex];
  const tgt = slides[idx];

  // Only body transitions, header/footer stay
  tgt.animate([
    { opacity: 0, transform: `translateX(30px)` },
    { opacity: 1, transform: 'translateX(0)' }
  ], { duration: 450, easing: 'ease-out' });
}
